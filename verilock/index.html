// Try sending 0.02 ETH without approval token
// Safe UI calls Guard.checkTransaction()
// → Guard reverts with TokenMissing()

Safe.sendTransaction({
  to: "0x090f...ED563",
  value: "0.02 ETH",
  data: "0x"  // No token! Guard will reject.
})

// Result: REVERTED
// Safe UI shows: "Simulation Error: Reverted"
// 1. Get approval from Notary
const token = await notary.approve({
  agent_id: "trading-bot-01",
  destination: "0x090f...ED563",
  amount_usd: 20,
  purpose: "defi_yield_optimization"
});

// 2. Encode token into data
const data = TollgateSDK.encodeTransaction(calldata, token);

// 3. Submit with token
await Safe.sendTransaction({ to, value, data });

// Result: SUCCESS
// Tollgate Guard — Solidity
function checkTransaction(...) external {
  // Only Safe can call
  require(msg.sender == safeAddress);

  // Extract token from calldata
  ApprovalTokenData memory token = _extractToken(data);

  // Verify Notary signature (EIP-712)
  address recovered = _recoverSigner(token);
  require(recovered == notaryAddress);

  // Check expiry, nonce, destination match...
}
